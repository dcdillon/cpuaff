326  inline bool get_affinity(cpu_set_type &cpus) const
327  {
328  cpus.clear();
329  std::set< cpu_identifier_wrapper_type > ids;
330 
331  if (get_affinity_type()(ids))
332  {
333  typename std::set< cpu_identifier_wrapper_type >::iterator i =
334  ids.begin();

343  }
344 
345  return true;
346  }

348  return false;
