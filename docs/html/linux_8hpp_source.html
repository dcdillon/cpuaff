94 struct cpu_loader
95 {
96  inline bool operator()(cpu_loader_vector_type &v)
97  {
98  v.clear();
99 
100  std::vector< sysfs_reader::pu > pus;
101  std::map< int32_t, std::map< int32_t, int32_t > > cores_by_socket;
102  std::map< int32_t, std::map< int32_t, std::vector< int32_t > > >
103  pus_by_socket_by_core;
104 
105  if (sysfs_reader::load_cpus(pus))
106  {
107  std::vector< sysfs_reader::pu >::iterator pu = pus.begin();
108  std::vector< sysfs_reader::pu >::iterator puend = pus.end();
109 
110  for (; pu != puend; ++pu)
111  {
112  core_type core;
113 
114  std::map< int32_t, int32_t >::iterator i =
115  cores_by_socket[pu->socket].find(pu->core);
116 
117  if (i != cores_by_socket[pu->socket].end())
118  {
119  core = core_type(i->second);
120  }
121  else
122  {
123  core = core_type(cores_by_socket[pu->socket].size());
124  cores_by_socket[pu->socket][pu->core] = int32_t(core);
125  }
126 
127  processing_unit_type pu_id;
128 
129  pu_id = pus_by_socket_by_core[pu->socket][core].size();
130  pus_by_socket_by_core[pu->socket][core].push_back(0);
131 
132  v.push_back(cpu_info(
133  cpu_spec(socket_type(pu->socket), core_type(core),
134  processing_unit_type(pu_id)),
135  cpu_identifier_type(pu->native), numa_type(pu->node)));
136  }
137  }
138 
139  return !!v.size();
140  }

143 struct get_affinity
144 {
145  inline bool operator()(std::set< cpu_identifier_wrapper > &cpus)
146  {
147  cpu_set_t cpu_set;
148  if (0 == sched_getaffinity(0, sizeof(cpu_set_t), &cpu_set))
149  {
150  for (cpu_identifier_type i = 0; i < CPU_SETSIZE; ++i)
151  {
152  if (CPU_ISSET(i, &cpu_set))
153  {
154  cpus.insert(cpu_identifier_wrapper(i));
155  }
156  }
157 
158  return true;
159  }
160  else
161  {
162  return false;
163  }
164  }

167 struct set_affinity
168 {
169  inline bool operator()(const std::set< cpu_identifier_wrapper > &cpus)
170  {
171  cpu_set_t cpu_set;
172  CPU_ZERO(&cpu_set);
173 
174  std::set< cpu_identifier_wrapper >::iterator i = cpus.begin();
175  std::set< cpu_identifier_wrapper >::iterator iend = cpus.end();
176 
177  for (; i != iend; ++i)
178  {
179  CPU_SET(i->get(), &cpu_set);
180  }
181 
182  return (0 == sched_setaffinity(0, sizeof(cpu_set_t), &cpu_set));
183  }
