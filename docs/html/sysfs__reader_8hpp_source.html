170 inline bool read_cpu(std::vector< pu > &pus, int32_t cpu)
171 {
172  int32_t socket = read_socket(cpu);
173  int32_t core = read_core(cpu);
174 
175  pu u;
176  u.node = -1;
177  u.native = cpu;
178  u.socket = (socket == -1) ? 0 : socket;
179  u.core = (core == -1) ? 0 : core;
180  pus.push_back(u);
181  return true;
182 }

185 {

187  int32_t core = read_core(node, cpu);

219 inline bool load_cpus(std::vector< pu > &pus)
220 {
221  pus.clear();
222 
223  std::set< int32_t > nodes;
224 
225  if (read_nodes(nodes))
226  {
227  std::set< int32_t >::iterator i = nodes.begin();
228  std::set< int32_t >::iterator iend = nodes.end();
229 
230  for (; i != iend; ++i)
231  {
232  read_node(pus, *i);
233  }
234  }
235  else
236  {
237  // we don't have nodes to read, so let's just set node to
238  // -1 for everything and read the cpus
239  std::set< int32_t > cpus;
240 
241  if (read_cpus(cpus))
242  {
243  std::set< int32_t >::iterator i = cpus.begin();
244  std::set< int32_t >::iterator iend = cpus.end();
245 
246  for (; i != iend; ++i)
247  {
248  read_cpu(pus, *i);
249  }
250  }
251  else
252  {
253  DIR *dir;
254  struct dirent *ent;
255 
256  if ((dir = opendir("/sys/devices/system/cpu")) != NULL)
257  {
258  while ((ent = readdir(dir)) != NULL)
259  {
260  std::string file = ent->d_name;
261 
262  if (file.substr(0, 3) == "cpu")
263  {
264  int32_t cpu = atoi(file.substr(3).c_str());
265  read_cpu(pus, cpu);
266  }
267  }
268  }
269  }
270  }
271 
272  return !!pus.size();
273 }
