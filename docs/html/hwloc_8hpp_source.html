113 class topology_reader
114 {
115  public:
116  inline topology_reader()
117  : numa_id_(0), socket_id_(0), core_id_(0), processing_unit_id_(0)
118  {
119  }
120 
121  inline void read()
122  {
123  read(hwloc_get_root_obj(topology::instance().get()), 0);
124  }
125 
126  inline std::vector< cpu_info > &cpus() { return cpus_; }
127 
128  private:
129  inline void read(hwloc_obj_t obj, int depth)
130  {
131  if (obj->type == HWLOC_OBJ_NODE)
132  {
133  numa_id_ = obj->logical_index;
134  }
135  else if (obj->type == HWLOC_OBJ_SOCKET)
136  {
137  socket_id_ = obj->logical_index;
138  core_id_ = -1;
139  }
140  else if (obj->type == HWLOC_OBJ_CORE)
141  {
142  ++core_id_;
143  processing_unit_id_ = 0;
144  }
145  else if (obj->type == HWLOC_OBJ_PU)
146  {
147  cpus_.push_back(
148  cpu_info(cpu_spec(socket_type(socket_id_), core_type(core_id_),
149  processing_unit_type(processing_unit_id_++)),
150  hwloc_bitmap_first(obj->cpuset), numa_type(numa_id_)));
151  }
152 
153  for (unsigned int i = 0; i < obj->arity; ++i)
154  {
155  read(obj->children[i], depth + 1);
156  }
157  }
158 
159  private:
160  unsigned int numa_id_;
161  unsigned int socket_id_;
162  unsigned int core_id_;
163  unsigned int processing_unit_id_;
164  std::vector< cpu_info > cpus_;
165 };

188 struct get_affinity
189 {
190  inline bool operator()(std::set< cpu_identifier_wrapper > &cpus)
191  {
192  bool retval = false;
193  hwloc_cpuset_t cpu_set = hwloc_bitmap_alloc();
194 
195  if (0 == hwloc_get_cpubind(topology::instance().get(), cpu_set,
196  HWLOC_CPUBIND_THREAD))
197  {
198  unsigned int id;
199  hwloc_bitmap_foreach_begin(id, cpu_set)
200  cpus.insert(cpu_identifier_wrapper(id));
201  hwloc_bitmap_foreach_end();
202 
203  retval = true;
204  }
205 
206  hwloc_bitmap_free(cpu_set);
207 
208  return retval;
209  }
210 };

212 struct set_affinity
213 {
214  inline bool operator()(const std::set< cpu_identifier_wrapper > &cpus)
215  {
216  hwloc_cpuset_t cpu_set = hwloc_bitmap_alloc();
217  hwloc_bitmap_zero(cpu_set);
218 
219  std::set< cpu_identifier_wrapper >::const_iterator i =
220  cpus.begin();
221  std::set< cpu_identifier_wrapper >::const_iterator iend =
222  cpus.end();
223 
224  for (; i != iend; ++i)
225  {
226  hwloc_bitmap_set(cpu_set, i->get());
227  }
228 
229  bool retval = (0 == hwloc_set_cpubind(topology::instance().get(),
230  cpu_set, HWLOC_CPUBIND_THREAD));
231  hwloc_bitmap_free(cpu_set);
232 
233  return retval;
234  }
