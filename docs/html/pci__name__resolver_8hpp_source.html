109  inline bool initialize(const std::string &file)
110  {
111  std::ifstream infile(file.c_str());
112  bool retval = infile.good();
113 
114  int32_t vendor = 0;
115  std::string vendor_description;
116  int32_t device = 0;
117  std::string device_description;
118 
119  while (infile.good())
120  {
121  std::string line;
122  std::getline(infile, line);
123 
124  if (line.length() > 2 && line[0] != '#' &&
125  line.substr(0, 2) != "C ")
126  {
127  if (line[0] == '\t')
128  {
129  if (line[1] != '\t')
130  {
131  if (parse_device(line, device, device_description))
132  {
133  description_by_id_[pci_device_spec(vendor,
134  device)] =
135  pci_device_description(vendor_description,
136  device_description);
137  }
138  else
139  {
140  device = 0;

142  }
143  }
144  }

158  device = 0;

160  }

164  return retval;
